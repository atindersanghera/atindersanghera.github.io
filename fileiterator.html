File Iterator

This python code iterates python commands over multiple files in a folder.

This example adds the name of the file into a dataframe.

from bs4 import BeautifulSoup as BS
import re
import pandas as pd
import os
path = r"source_path"
files = os.listdir(path)
df = pd.DataFrame([], columns = ['name'])
for file in files:
file_path = os.path.join(path,file)
f = open(file_path, "r")
# Do what you want with the file here :)
f.close()
df = df.append({'name': file}, ignore_index=True)
df.to_csv (r"destination_path", index = False, header=True)
